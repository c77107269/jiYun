			2、字符串的拼接
				console.log(11 + 11);
				console.log('hello' + ' world');
				console.log('100' + '100');
				console.log('11' + 11);
				console.log('male:' + true);
			3、字符串的方法
				concat() // 用于连接两个或者多个字符串，功能和'+'拼接相同
				split() // 用于将字符串分割成字符串数组
				indexOf和lastIndexOf // 检索字符串
				slice() // 提取字符串的某个部分，并以新的字符串返回被提取的部分
				substring() // 用于提取字符串中介于两个指定下标之间的字符,如果只有一个参数，会返回的指定字符串到末尾
				substr() // 可在字符串中抽取从指定下标开始的指定数量的字符，如果只有一个参数，会返回的指定字符串到末尾
				charAt() // 获取指定位置的字符
				replace() // 替换字符串中的一些字符（只能替换第一个），或者替换一个和正则表达式相匹配的内容
				trim()  // 用来删除字符串前后的空格
				toUpperCase() // 把字符串转为大写，返回新的字符串
				toLowerCase() // 把字符串转为小写，返回新的字符串
				includes() // 判断字符串中是否存在某一个字符串
